Particle Engine Code

                            
                                import pygame
                                import sys
                                import math
                                import random
                                
                                pygame.init()
                                
                                WIDTH, HEIGHT = 1600, 900
                                screen = pygame.display.set_mode((WIDTH, HEIGHT))
                                pygame.display.set_caption("Pygame Physics - Multiple Interacting Balls with Angle-Based Bounce")
                                
                                BLACK = (0, 0, 0)
                                GREEN = (0, 255, 0)
                                WHITE = (255, 255, 255)
                                
                                border_thickness = 7
                                border_offset = 50
                                
                                left = border_offset + border_thickness
                                right = WIDTH - (border_offset + border_thickness)
                                top = border_offset + border_thickness
                                bottom = HEIGHT - (border_offset + border_thickness)
                                
                                circle_radius = 7 #17 #7
                                gravity = 0.175 #0.1 #0.175
                                bounce_damping = 0.5
                                friction = 1
                                
                                num_particles = 425 #250 #425
                                spawn_top = top + 100
                                spawn_bottom = HEIGHT // 2
                                
                                balls = []
                                for _ in range(num_particles):
                                    x = random.randint(left + circle_radius, right - circle_radius)
                                    y = random.randint(spawn_top, spawn_bottom)
                                    balls.append({
                                        "x": x,
                                        "y": y,
                                        "velocity_x": random.uniform(-1, 1),
                                        "velocity_y": random.uniform(-1, 1)
                                    })
                                
                                repel_radius = 200
                                repel_force = 7
                                
                                suction_radius = 200
                                suction_force = 7
                                
                                clock = pygame.time.Clock()
                                running = True
                                
                                font = pygame.font.SysFont(None, 30)
                                
                                def get_color_from_speed(speed, max_speed=10):
                                    normalized_speed = min(speed / max_speed, 1)
                                
                                    if normalized_speed < 0.25:
                                        red = 0
                                        green = int(255 * (normalized_speed / 0.25))
                                        blue = 255 - int(255 * (normalized_speed / 0.25))
                                    elif normalized_speed < 0.5:
                                        red = int(255 * ((normalized_speed - 0.25) / 0.25))
                                        green = 255
                                        blue = 0
                                    elif normalized_speed < 0.75:
                                        red = 255
                                        green = 255 - int(90 * ((normalized_speed - 0.5) / 0.25))
                                        blue = 0
                                    else:
                                        red = 255
                                        green = 165 - int(165 * ((normalized_speed - 0.75) / 0.25))
                                        blue = 0
                                
                                    color = (max(0, min(255, red)), max(0, min(255, green)), max(0, min(255, blue)))
                                    return color
                                
                                while running:
                                    screen.fill(BLACK)
                                    
                                    pygame.draw.rect(screen, GREEN, (border_offset, border_offset,
                                                                     WIDTH - 2 * border_offset,
                                                                     HEIGHT - 2 * border_offset), border_thickness)
                                
                                    mouse_x, mouse_y = pygame.mouse.get_pos()
                                    mouse_buttons = pygame.mouse.get_pressed()
                                
                                    fps = int(clock.get_fps())
                                    fps_text = font.render(f"FPS: {fps}", True, WHITE)
                                    screen.blit(fps_text, (WIDTH - fps_text.get_width() - 10, 10))
                                
                                    for ball in balls:
                                        ball["velocity_y"] += gravity
                                        ball["y"] += ball["velocity_y"]
                                        ball["x"] += ball["velocity_x"]
                                
                                        if ball["y"] + circle_radius >= bottom:
                                            ball["velocity_x"] *= friction
                                            ball["y"] = bottom - circle_radius
                                            ball["velocity_y"] = -ball["velocity_y"] * bounce_damping
                                
                                        if ball["y"] - circle_radius <= top:
                                            ball["y"] = top + circle_radius
                                            ball["velocity_y"] = -ball["velocity_y"] * bounce_damping
                                
                                        if ball["x"] - circle_radius <= left:
                                            ball["x"] = left + circle_radius
                                            ball["velocity_x"] = -ball["velocity_x"] * bounce_damping
                                
                                        if ball["x"] + circle_radius >= right:
                                            ball["x"] = right - circle_radius
                                            ball["velocity_x"] = -ball["velocity_x"] * bounce_damping
                                
                                    if mouse_buttons[0]:
                                        for ball in balls:
                                            dx = ball["x"] - mouse_x
                                            dy = ball["y"] - mouse_y
                                            distance = math.hypot(dx, dy)
                                            if distance < repel_radius:
                                                angle = math.atan2(dy, dx)
                                                ball["velocity_x"] += math.cos(angle) * repel_force * random.uniform(0.5, 1.5)
                                                ball["velocity_y"] += math.sin(angle) * repel_force * random.uniform(0.5, 1.5)
                                
                                    if mouse_buttons[2]: 
                                        for ball in balls:
                                            dx = mouse_x - ball["x"]
                                            dy = mouse_y - ball["y"]
                                            distance = math.hypot(dx, dy)
                                            if distance < suction_radius:
                                                angle = math.atan2(dy, dx)
                                                ball["velocity_x"] += math.cos(angle) * suction_force * random.uniform(0.5, 1.5)
                                                ball["velocity_y"] += math.sin(angle) * suction_force * random.uniform(0.5, 1.5)
                                
                                    for i in range(len(balls)):
                                        for j in range(i + 1, len(balls)):
                                            dx = balls[j]["x"] - balls[i]["x"]
                                            dy = balls[j]["y"] - balls[i]["y"]
                                            distance = math.hypot(dx, dy)
                                
                                            if distance < 2 * circle_radius:
                                                angle = math.atan2(dy, dx)
                                
                                                v1 = balls[i]["velocity_x"] * math.cos(angle) + balls[i]["velocity_y"] * math.sin(angle)
                                                v2 = balls[j]["velocity_x"] * math.cos(angle) + balls[j]["velocity_y"] * math.sin(angle)
                                                balls[i]["velocity_x"] += (v2 - v1) * math.cos(angle)
                                                balls[i]["velocity_y"] += (v2 - v1) * math.sin(angle)
                                                balls[j]["velocity_x"] += (v1 - v2) * math.cos(angle)
                                                balls[j]["velocity_y"] += (v1 - v2) * math.sin(angle)
                                
                                                overlap = 2 * circle_radius - distance
                                                correction = overlap / 2
                                                balls[i]["x"] -= correction * math.cos(angle)
                                                balls[i]["y"] -= correction * math.sin(angle)
                                                balls[j]["x"] += correction * math.cos(angle)
                                                balls[j]["y"] += correction * math.sin(angle)
                                
                                    for ball in balls:
                                        speed = math.hypot(ball["velocity_x"], ball["velocity_y"])
                                        color = get_color_from_speed(speed)
                                        pygame.draw.circle(screen, color, (int(ball["x"]), int(ball["y"])), circle_radius)
                                
                                    for event in pygame.event.get():
                                        if event.type == pygame.QUIT:
                                            running = False
                                
                                    pygame.display.flip()
                                    clock.tick(120)
                                
                                pygame.quit()
                                sys.exit()

If anyone gives a monkey about the maths behind all this, i did it really simply im not smart enough to use crazy equations with itegration and differentials so basically the entirety of the maths is the pythagorean theorum used to calculator velocities between boundaries and collisions and basic trig for calculating the angle in which particles bounce :)
